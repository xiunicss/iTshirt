// script 로 호출
function designAlert(){
    UI.alert({
        title : 'alert 제목', // 생략 가능
        innerHtml : '텍스트나 html 태그 입력',
        cancel : true, // default cancel 버튼 없음
        cancelTxt : '아니요', // 취소 버튼 텍스트 변경
        confirmTxt : '예', // 확인버튼 텍스트 변경
        onCancel : function(){
            console.log( '알럿 취소 클릭' );
        },
        onConfirm : function(){
            console.log( '알럿 확인 클릭' );
        }
    });
}
